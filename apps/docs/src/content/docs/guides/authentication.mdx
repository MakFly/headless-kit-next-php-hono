## Login

### Frontend (Server Action)

```typescript
// apps/web/src/actions/auth.ts
'use server';

import { bff } from '@/lib/bff';
import { redirect } from 'next/navigation';

export async function loginAction(formData: FormData) {
  const email = formData.get('email') as string;
  const password = formData.get('password') as string;

  const result = await bff.post('/api/v1/auth/login', {
    email,
    password,
  });

  // Tokens are automatically stored in HttpOnly cookies by the proxy
  redirect('/dashboard');
}
```

### Frontend (Client Component)

```tsx
// apps/web/src/components/login-form.tsx
'use client';

import { loginAction } from '@/actions/auth';

export function LoginForm() {
  return (
    <form action={loginAction}>
      <input name="email" type="email" required />
      <input name="password" type="password" required />
      <button type="submit">Sign In</button>
    </form>
  );
}
```

<Aside type="caution">
  Never use `fetch()` directly in server actions. Always use the BFF client helpers from `@/lib/actions/_shared/bff-client.ts` (`bffGet`, `bffPost`, etc.). They handle cookie forwarding, which is not supported by `credentials: 'include'` in Node.js server-side contexts.
</Aside>

## Registration

```typescript
'use server';

import { bff } from '@/lib/bff';

export async function registerAction(formData: FormData) {
  const result = await bff.post('/api/v1/auth/register', {
    name: formData.get('name') as string,
    email: formData.get('email') as string,
    password: formData.get('password') as string,
    password_confirmation: formData.get('password_confirmation') as string,
  });

  redirect('/dashboard');
}
```

## Logout

```typescript
'use server';

import { bff } from '@/lib/bff';
import { redirect } from 'next/navigation';

export async function logoutAction() {
  await bff.post('/api/v1/auth/logout');
  // The proxy clears all auth cookies automatically
  redirect('/auth/login');
}
```

## Get Current User

```typescript
'use server';

import { bff } from '@/lib/bff';
// User type is defined inline in each app

export async function getCurrentUser() {
  try {
    const response = await bff.get('/api/v1/auth/me');
    return response.user;
  } catch {
    return null;
  }
}
```

## Token Storage

Tokens are stored in three HttpOnly cookies managed by the BFF proxy:

| Cookie | HttpOnly | Content |
| ------ | -------- | ------- |
| `auth_token_{backend}` | Yes | Access token (1 hour) |
| `refresh_token_{backend}` | Yes | Refresh token (30 days) |
| `token_expires_at_{backend}` | No | ISO 8601 expiration timestamp |

The `token_expires_at` cookie is readable by client-side JavaScript, enabling proactive refresh without exposing the actual tokens.

## Proactive Token Refresh

The BFF middleware handles token refresh automatically. You can also check expiration client-side:

```typescript
// These helpers are available in the app's token service

function getExpiresCookie(backend: string): string | null {
  const match = document.cookie.match(
    new RegExp(`token_expires_at_${backend}=([^;]+)`)
  );
  return match ? decodeURIComponent(match[1]) : null;
}

export function checkAuthStatus(backend: string) {
  const expiresAt = getExpiresCookie(backend);

  if (!expiresAt || isTokenExpiredByTimestamp(expiresAt)) {
    return 'expired';
  }

  if (shouldRefreshByTimestamp(expiresAt)) {
    return 'needs-refresh';
  }

  return 'valid';
}
```

The middleware uses two refresh strategies:

1. **Proactive**: When a request hits the middleware and the token is within 5 minutes of expiration, it sets `x-bff-refresh-needed: true`. The proxy handler refreshes the token before forwarding the request.

2. **Reactive**: If the backend returns 401, the proxy attempts a refresh and retries the request. If the refresh also fails, the user is logged out.

## BFF Client Setup

Configure the BFF client once and reuse across all server actions:

```typescript
// apps/web/src/lib/actions/_shared/bff-client.ts
// Provides bffGet, bffPost, bffPut, bffPatch, bffDelete
// Usage in server actions:
'use server';
import { bffGet, bffPost } from '@/lib/actions/_shared/bff-client';

const user = await bffGet('/api/v1/auth/me');
await bffPost('/api/v1/auth/logout');
```

## Error Handling

The BFF client throws `BffActionError` on failure:

```typescript
import { BffActionError } from '@/lib/actions/_shared/errors';

try {
  await bff.post('/api/v1/auth/login', credentials);
} catch (error) {
  if (error instanceof BffActionError) {
    console.error(error.statusCode); // 401, 422, etc.
    console.error(error.code);       // 'UNAUTHORIZED', 'VALIDATION_ERROR'
    console.error(error.details);    // { email: ['Invalid credentials'] }
  }
}
```
